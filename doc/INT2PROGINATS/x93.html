1.3. Makefileのひな形

次に示すMakefileのひな形はATSプログラムをコンパイルする独自のMakefileを作るのに役立つでしょう。 またこのひな形は オンライン から取得できます。 あなたが make ユーティリティに習熟していなかったとしても、 オンラインからその学習のための情報を数多く見つけられるでしょう。 

######
#
# インストール環境によっては以下のように読みかえる必要があります。
#
# atscc -> patscc
# atsopt -> patsopt
# ATSHOME -> PATSHOME
#
######

ATSHOMEQ="$(ATSHOME)"

######

ATSCC=$(ATSHOMEQ)/bin/atscc
ATSOPT=$(ATSHOMEQ)/bin/atsopt

######

#
# HX: どれかの行のコメントを外してください。もしくは読み飛ばすこともできます。
#
ATSCCFLAGS=
#ATSCCFLAGS=-O2
#
# '-flto' はリンク時最適化を有効にします。ライブラリ関数のインライン展開などを行ないます。
#
#ATSCCFLAGS=-O2 -flto
#

######

cleanall::

######

#
# 次の3行のコメントを外してください。
# さらに [foo] という名前をあなたがコンパイルしたい名前に変更してください。
#

# foo: foo.dats
#   $(ATSCC) $(ATSCCFLAGS) -o $@ $< || echo $@ ": ERROR!!!"
# cleanall:: ; $(RMF) foo

######

#
# もしかすると、これらのルールも役に立つかもしれません。
#

# %_sats.o: %.sats ; $(ATSCC) $(ATSCCFLAGS) -c $< || echo $@ ": ERROR!!!"
# %_dats.o: %.dats ; $(ATSCC) $(ATSCCFLAGS) -c $< || echo $@ ": ERROR!!!"

######

RMF=rm -f

######

clean:: ; $(RMF) *~
clean:: ; $(RMF) *_?ats.o
clean:: ; $(RMF) *_?ats.c

cleanall:: clean

###### end of [Makefile] ######
